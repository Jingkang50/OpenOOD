Open Models

Check out the available models and our Colab tutorials. 
# !pip install git+https://github.com/OpenOOD/OpenOOD@v0.2.1
from OpenOOD.utils import load_model
# Load a model from the Open Models
model = load_model(model_name='Rebuffi2021Fixing_70_16_cutmix_extra',
                   dataset='cifar10',
                   threat_model='Linf')

# Evaluate the OOD of the model using AutoAttack
from OpenOOD.eval import benchmark
clean_acc, near_ood_auroc, far_ood_auroc = benchmark(model,
                                  dataset='cifar10',
                                  threat_model='Linf')
